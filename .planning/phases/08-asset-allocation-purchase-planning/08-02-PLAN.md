       ```dart
       @DriftAccessor(tables: [AllocationTargets])
       class AllocationTargetDao extends DatabaseAccessor<AppDatabase> with _$AllocationTargetDaoMixin {
         AllocationTargetDao(super.db);

         Stream<List<AllocationTarget>> watchForProfile(String lifeProfileId) =>
           (select(allocationTargets)..where((t) => t.lifeProfileId.equals(lifeProfileId)))
             .watch();

         Future<List<AllocationTarget>> getForProfile(String lifeProfileId) =>
           (select(allocationTargets)..where((t) => t.lifeProfileId.equals(lifeProfileId)))
             .get();

         Future<void> upsertTarget(AllocationTargetsCompanion target) =>
           into(allocationTargets).insertOnConflictUpdate(target);

         Future<void> replaceAllForProfile(String lifeProfileId, List<AllocationTargetsCompanion> targets) async {
           await transaction(() async {
             await (delete(allocationTargets)..where((t) => t.lifeProfileId.equals(lifeProfileId))).go();
             for (final t in targets) {
               await into(allocationTargets).insert(t);
             }
           });
         }
       }
       ```

    2. Create lib/core/database/daos/decision_dao.dart:
       ```dart
       @DriftAccessor(tables: [Decisions])
       class DecisionDao extends DatabaseAccessor<AppDatabase> with _$DecisionDaoMixin {
         DecisionDao(super.db);

         Stream<List<Decision>> watchForUser(String userId, String familyId) =>
           (select(decisions)
             ..where((d) => d.userId.equals(userId) & d.familyId.equals(familyId) & d.deletedAt.isNull())
             ..orderBy([(d) => OrderingTerm.desc(d.createdAt)]))
             .watch();

         Future<Decision?> getById(String id) =>
           (select(decisions)..where((d) => d.id.equals(id))).getSingleOrNull();

         Future<void> insertDecision(DecisionsCompanion decision) =>
           into(decisions).insert(decision);

         Future<void> markImplemented(String id) =>
           (update(decisions)..where((d) => d.id.equals(id)))
             .write(DecisionsCompanion(
               status: const Value('implemented'),
               implementedAt: Value(DateTime.now()),
             ));

         Future<void> softDelete(String id) =>
           (update(decisions)..where((d) => d.id.equals(id)))
             .write(DecisionsCompanion(deletedAt: Value(DateTime.now())));
       }
       ```
