    ```dart
    // Stream of allocation rules for family, sorted by priority
    final allocationRulesProvider = StreamProvider.family<List<SavingsAllocationRule>, String>(
      (ref, familyId) {
        final db = ref.read(appDatabaseProvider);
        return SavingsAllocationRuleDao(db).watchByFamily(familyId);
      },
    );

    // Current month surplus in paise from MonthlyMetrics
    final monthlySurplusProvider = FutureProvider.family<int, String>(
      (ref, familyId) async {
        final db = ref.read(appDatabaseProvider);
        final dao = MonthlyMetricsDao(db);
        final now = DateTime.now();
        final monthKey = '${now.year}-${now.month.toString().padLeft(2, '0')}';
        final metrics = await dao.getByMonth(familyId, monthKey);
        if (metrics == null) return 0;
        return metrics.totalIncomePaise - metrics.totalExpensesPaise;
      },
    );

    // Advisory allocation output
    final allocationAdvisoryProvider = FutureProvider.family<List<AllocationAdvice>, String>(
      (ref, familyId) async {
        final db = ref.read(appDatabaseProvider);
        final ruleDao = SavingsAllocationRuleDao(db);
        final accountDao = AccountDao(db);

        // 1. Get rules sorted by priority
        final dbRules = await ruleDao.getByFamily(familyId);
        final rules = dbRules.map((r) => AllocationRule(
          priority: r.priority,
          targetType: r.targetType,
          targetId: r.targetId,
          allocationType: r.allocationType,
          amountPaise: r.amountPaise,
          percentageBp: r.percentageBp,
        )).toList();

        // 2. Get surplus
        final surplus = await ref.read(monthlySurplusProvider(familyId).future);
        if (surplus <= 0) return [];

        // 3. Build targets map
        // Key format: '$targetType:$targetId' (targetId is '' if null)
        final targets = <String, AllocationTarget>{};

        // Emergency fund target
        // Read EF state from existing provider chain
        final efAccounts = await accountDao.watchEmergencyFundAccounts(familyId).first;
        final efBalance = efAccounts.fold<int>(0, (sum, a) => sum + a.balance);
        // Get EF target from life profile (read existing emergencyFundStateProvider pattern)
        // For now, add EF target if EF accounts exist
        // ... (executor reads emergencyFundStateProvider to get target amount)

        // Sinking fund targets
        final goalDao = GoalDao(db);
        final sfGoals = await goalDao.watchByCategories(familyId, ['sinkingFund']).first;
        for (final goal in sfGoals) {
          final key = 'sinkingFund:${goal.id}';
          targets[key] = AllocationTarget(
            targetType: 'sinkingFund',
            targetId: goal.id,
            targetName: goal.name,
            currentPaise: goal.currentAmount ?? 0,
            targetPaise: goal.targetAmount,
          );
        }

        // Investment goal targets
        final igGoals = await goalDao.watchByCategories(familyId, ['investmentGoal']).first;
        for (final goal in igGoals) {
          final key = 'investmentGoal:${goal.id}';
          targets[key] = AllocationTarget(
            targetType: 'investmentGoal',
            targetId: goal.id,
            targetName: goal.name,
            currentPaise: goal.currentAmount ?? 0,
            targetPaise: goal.targetAmount,
          );
        }

        // Opportunity fund target
        final oppAccount = await accountDao.watchOpportunityFund(familyId).first;
        if (oppAccount != null) {
          targets['opportunityFund:'] = AllocationTarget(
            targetType: 'opportunityFund',
            targetId: null,
            targetName: oppAccount.name,
            currentPaise: oppAccount.balance,
            targetPaise: oppAccount.opportunityFundTargetPaise ?? 0, // 0 = unlimited
          );
        }

        // 4. Distribute
        return SavingsAllocationEngine.distribute(
          surplusPaise: surplus,
          rules: rules,
          targets: targets,
        );
      },
    );
    ```
