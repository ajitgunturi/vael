    ```dart
      final String targetName;
      final int currentPaise;

    }

    class AllocationAdvice {
      final String targetType;
      final String? targetId;
      final String targetName;
      final int allocatedPaise;
      final int remainingToTarget;  // targetPaise - currentPaise - allocatedPaise (0 if unlimited)
    }
    ```

    Engine:
    ```dart
    class SavingsAllocationEngine {
      SavingsAllocationEngine._();

      static List<AllocationAdvice> distribute({
        required int surplusPaise,
        required List<AllocationRule> rules,         // MUST be sorted by priority ascending
        required Map<String, AllocationTarget> targets,  // keyed by '$targetType:$targetId'
      }) {
        // 1. var remaining = surplusPaise
        // 2. For each rule in priority order:
        //    a. Find target by key '${rule.targetType}:${rule.targetId}'
        //    b. If target not found, skip
        //    c. Compute gap = target.targetPaise == 0 ? remaining : max(0, target.targetPaise - target.currentPaise)
        //    d. If gap <= 0, skip (target met)
        //    e. Compute ruleAmount: fixed -> rule.amountPaise!, percentage -> surplusPaise * rule.percentageBp! ~/ 10000
        //    f. allocated = min(ruleAmount, remaining, gap)
        //    g. remaining -= allocated
        //    h. Add AllocationAdvice(targetType, targetId, targetName, allocated, gap - allocated)
        // 3. Return advice list
      }
    }
    ```

    Use `import 'package:collection/collection.dart'` for groupBy in CashFlowEngine.
