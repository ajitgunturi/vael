  <action>
    1. Create lib/core/financial/insights_engine.dart following pure static pattern (private constructor):
       - enum InsightSeverity { critical, warning, info }
       - enum InsightType { efBelowTarget, allocationOffTarget, fiDateSlipping, sinkingFundUnderfunded }
       - class PlanningInsight { final InsightType type; final InsightSeverity severity; final String title; final String description; }
       - static PlanningInsight? efBelowTarget({required double coverageMonths, required int targetMonths})
         - Returns null if coverageMonths >= targetMonths
         - severity=critical if coverageMonths < targetMonths * 0.5
         - severity=warning otherwise
         - title: "Emergency fund below target"
         - description: "{X.X} months short of {Y}-month target"
       - static PlanningInsight? allocationOffTarget({required List<RebalancingDelta> deltas})
         - Find max absolute (actualBp - targetBp) across deltas
         - Returns null if maxDeviation <= 500
         - severity=critical if maxDeviation > 1500
         - severity=warning otherwise
         - title: "Asset allocation off-target"
         - description: "{assetClass} is {X.X}% off target"
       - static PlanningInsight? fiDateSlipping({required int? currentYearsToFi, required int? previousYearsToFi})
         - Returns null if either is null or current <= previous
         - severity=warning always
         - title: "FI date slipping"
         - description: "FI timeline extended by {N} year(s)"
       - static PlanningInsight? sinkingFundUnderfunded({required String paceStatus, required String fundName, required int monthlyNeededPaise, required int monthlyActualPaise})
         - Returns null if paceStatus == 'onTrack'
         - severity=critical if paceStatus == 'atRisk'
         - severity=warning if paceStatus == 'behind'
         - title: "{fundName} underfunded"
         - description: "Need {X}/mo, currently saving {Y}/mo"
